				jwplayer("newsVid").onReady(function(){
					function toggleCaption(evt) {
						if (evt.keyCode == 67 && document.activeElement.id == "newsVid") {
							var capLength = jwplayer("newsVid").getCaptionsList().length;
							var nextCap = jwplayer("newsVid").getCurrentCaptions()+1;
							if (capLength > 1) {
								if(nextCap > capLength-1) {
									jwplayer("newsVid").setCurrentCaptions(0);
								}
								else {
									jwplayer("newsVid").setCurrentCaptions(nextCap);
								}
							}          
						}
					}
					window.onkeydown = toggleCaption;
					document.getElementById("goto_newsVid").addEventListener("click",function(){
						document.getElementById("newsVid").focus();
